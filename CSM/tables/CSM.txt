CREATE TABLE CSMUSER (
  ID CHAR(38) NOT NULL,
  CONTEXTID CHAR(38) NOT NULL,
  NAME VARCHAR(60) NOT NULL,
  DESCRIPTION TEXT NULL,
  CSMUSERNAME VARCHAR(15) NOT NULL,
  PROFILEID CHAR(38) NOT NULL,
  LASTLOGIN DATETIME NULL,
  FL_ACTIVE   BOOL NULL,
  CREATEDAT DATETIME NOT NULL,
  CREATEDBY DATETIME NOT NULL,
  CONTACTID  CHAR(38) NULL,
  ACCESSCONTROLID CHAR(38) NOT NULL,
  PRIMARY KEY(ID)
);




CREATE TABLE CONTACT (
  ID CHAR(38) NOT NULL AUTO_INCREMENT,
  AREA VARCHAR(60) NULL,
  PHONE VARCHAR(60) NULL,
  MOBILE VARCHAR(60) NULL,
  EMAIL VARCHAR(60) NULL,
  PRIMARY KEY(ID)
);




CREATE TABLE USERACCESSCONTROL (
  ID CHAR(38) NOT NULL AUTO_INCREMENT,
  USERPASSWORD CHAR(32) NULL,
  EXPIRATIONDATE DATE NULL,
  LASTCHANGEDATE DATE NULL,
  FL_CHANGENEXTLOGIN BOOL NULL,
  PRIMARY KEY(ID)
);


CREATE TABLE CSMDEVICE (
  ID CHAR(38) NOT NULL,
  NAME VARCHAR(60) NOT NULL AUTO_INCREMENT,
  DESCRIPTION TEXT NULL,
  CONTEXTID CHAR(38) NOT NULL,
  PRIMARY KEY(ID)
)
TYPE=InnoDB;





CREATE TABLE USERPROFDEVICEDENIED (
  PRFID CHAR(38) NOT NULL AUTO_INCREMENT,
  DEVID CHAR(38) NULL,
  PRIMARY KEY(PRFID)
)
TYPE=InnoDB;




CREATE TABLE CSMDEVICE (
  ID CHAR(38) NOT NULL,
  NAME VARCHAR(60) NOT NULL ,
  DESCRIPTION TEXT NULL,
  PRIMARY KEY(ID)
)
TYPE=InnoDB;
